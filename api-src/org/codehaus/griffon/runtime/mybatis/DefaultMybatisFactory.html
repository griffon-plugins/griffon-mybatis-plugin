059 public class DefaultMybatisFactory extends AbstractObjectFactory<SqlSessionFactory> implements MybatisFactory {
060     private static final Logger LOG = LoggerFactory.getLogger(DefaultMybatisFactory.class);
061     private static final String ERROR_SESSION_FACTORY_NAME_BLANK = "Argument 'sessionFactoryName' must not be blank";
062     private final Set<String> sessionFactoryNames = new LinkedHashSet<>();
063

065     private DataSourceFactory dataSourceFactory;

084     @Inject
085     public DefaultMybatisFactory(@Nonnull @Named("mybatis") griffon.core.Configuration configuration, @Nonnull GriffonApplication application) {
086         super(configuration, application);
087         sessionFactoryNames.add(KEY_DEFAULT);
088 
089         if (configuration.containsKey(getPluralKey())) {
090             Map<String, Object> sessionFactories = (Map<String, Object>) configuration.get(getPluralKey());
091             sessionFactoryNames.addAll(sessionFactories.keySet());
092         }
093     }

096     @Override
097     public Set<String> getSessionFactoryNames() {
098         return sessionFactoryNames;

103     public Map<String, Object> getConfigurationFor(@Nonnull String sessionFactoryName) {
104         requireNonBlank(sessionFactoryName, ERROR_SESSION_FACTORY_NAME_BLANK);
105         return narrowConfig(sessionFactoryName);
106     }

117         return "sessionFactories";

120     @Nonnull
121     @Override
122     public SqlSessionFactory create(@Nonnull String name) {
123         Map<String, Object> config = narrowConfig(name);
124         event("MybatisConnectStart", asList(name, config));
125         SqlSessionFactory sqlSessionFactory = createSqlSessionFactory(config, name);
126 
127         if (getConfigValueAsBoolean(config, "jmx", true)) {
128             sqlSessionFactory = new JMXAwareSqlSessionFactory(sqlSessionFactory);
129             registerMBeans(name, (JMXAwareSqlSessionFactory) sqlSessionFactory);
130         }
131 
132         try (SqlSession session = openSession(name, sqlSessionFactory)) {
133             for (Object o : injector.getInstances(MybatisBootstrap.class)) {
134                 ((MybatisBootstrap) o).init(name, session);
135             }
136         }
137 
138         event("MybatisConnectEnd", asList(name, config, sqlSessionFactory));
139         return sqlSessionFactory;
140     }
141 
142     @Override
143     public void destroy(@Nonnull String name, @Nonnull SqlSessionFactory instance) {
144         requireNonNull(instance, "Argument 'instance' must not be null");
145         Map<String, Object> config = narrowConfig(name);
146         event("MybatisDisconnectStart", asList(name, config, instance));
147 
148         try (SqlSession session = openSession(name, instance)) {
149             for (Object o : injector.getInstances(MybatisBootstrap.class)) {
150                 ((MybatisBootstrap) o).destroy(name, session);
151             }
152         }
153 
154         closeDataSource(name);
155 
156         if (getConfigValueAsBoolean(config, "jmx", true)) {
157             ((JMXAwareSqlSessionFactory) instance).disposeMBeans();
158         }
159 
160         event("MybatisDisconnectEnd", asList(name, config));
161     }
162 
163     private void registerMBeans(@Nonnull String name, @Nonnull JMXAwareSqlSessionFactory sqlSessionFactory) {
164         RecordingSqlSessionFactory recordingSqlSessionFactory = (RecordingSqlSessionFactory) sqlSessionFactory.getDelegate();
165         SqlSessionFactoryMonitor sqlSessionFactoryMonitor = new SqlSessionFactoryMonitor(metadata, recordingSqlSessionFactory, name);
166         sqlSessionFactory.addObjectName(mBeanManager.registerMBean(sqlSessionFactoryMonitor, false).getCanonicalName());
167     }
168 
169     @Nonnull
170     @SuppressWarnings("ConstantConditions")
171     private SqlSessionFactory createSqlSessionFactory(@Nonnull Map<String, Object> config, @Nonnull String dataSourceName) {
172         DataSource dataSource = getDataSource(dataSourceName);
173         Environment environment = new Environment(dataSourceName, new JdbcTransactionFactory(), dataSource);
174         Configuration configuration = new Configuration(environment);
175 
176         Map<String, Object> copyOfConfig = new LinkedHashMap<>(config);
177         copyOfConfig.remove("connect_on_startup");
178         GriffonClassUtils.setProperties(configuration, copyOfConfig);
179 
180         if (mappers.isEmpty()) {
181             readMappers();
182         }
183 
184         for (Class<?> mapper : mappers) {
185             configuration.addMapper(mapper);
186         }
187 
188         return new RecordingSqlSessionFactory(new SqlSessionFactoryBuilder().build(configuration));
189     }
190 
191     private void closeDataSource(@Nonnull String dataSourceName) {
192         DataSource dataSource = dataSourceStorage.get(dataSourceName);
193         if (dataSource != null) {
194             dataSourceFactory.destroy(dataSourceName, dataSource);
195             dataSourceStorage.remove(dataSourceName);
196         }
197     }
198 
199     @Nonnull
200     private DataSource getDataSource(@Nonnull String dataSourceName) {
201         DataSource dataSource = dataSourceStorage.get(dataSourceName);
202         if (dataSource == null) {
203             dataSource = dataSourceFactory.create(dataSourceName);
204             dataSourceStorage.set(dataSourceName, dataSource);
205         }
206         return dataSource;
207     }

209     @Nonnull
210     protected SqlSession openSession(@Nonnull String sessionFactoryName, @Nonnull SqlSessionFactory sqlSessionFactory) {
211         return sqlSessionFactory.openSession(true);
212     }
213 
214     private void readMappers() {
215         ServiceLoaderUtils.load(applicationClassLoader.get(), "META-INF/types/", MybatisMapper.class, new ServiceLoaderUtils.LineProcessor() {
216             @Override
217             public void process(@Nonnull ClassLoader classLoader, @Nonnull Class<?> type, @Nonnull String line) {
218                 try {
219                     Class<?> mapperClass = loadClass(line.trim(), classLoader);
220                     LOG.debug("Registering {}as mybatis mapper class", mapperClass.getName());
221                     mappers.add(mapperClass);
222                 } catch (Exception e) {
223                     if (LOG.isWarnEnabled()) {
224                         LOG.warn("Could not load " + type.getName() + " with " + line, sanitize(e));
225                     }
226                 }
227             }
228         });
229     }
230 
231     protected Class<?> loadClass(@Nonnull String className, @Nonnull ClassLoader classLoader) throws ClassNotFoundException {
232         ClassNotFoundException cnfe;
233 
234         ClassLoader cl = DefaultMybatisFactory.class.getClassLoader();
235         try {
236             return cl.loadClass(className);
237         } catch (ClassNotFoundException e) {
238             cnfe = e;
239         }
240 
241         cl = classLoader;
242         try {
243             return cl.loadClass(className);
244         } catch (ClassNotFoundException e) {
245             cnfe = e;
246         }
247 
248         throw cnfe;
249     }
